# 將資料拆成input X跟output Y
def create_sequences(data, seq_length):
    xs = []
    ys = []

    for i in range(len(data)-seq_length ):
        x = data.iloc[i:(i+seq_length)].to_numpy()
        y = data.iloc[i+seq_length]
        xs.append(np.reshape(x, (seq_length,1)))
        ys.append(y)

    return np.array(xs), np.array(ys)

class RV_dataset(Dataset):
    def __init__(self, train_test_split_ratio ,window_size, add_IV_input, mode='train'):
        self.mode = mode
        self.window_size = window_size
        self.tt_ratio = train_test_split_ratio
        self.add_IV_input = add_IV_input
        
        ''' 讀取RV,IV,GARCH等資料'''
        db_path =  '/Users/robert/Documents/碩士畢業論文/Code_and_db/RV_predict_project.db'
        conn = sqlite3.connect(db_path, timeout = 20)    
        df = pd.read_sql_query("SELECT * FROM IV_RV_GARCH_table", conn)  
        
        ''' 將波動度都*10 方便模型訓練 '''
        #df['daily_RV'] = df['daily_RV'] * 10
        #df['IV加權'] = df['IV加權'] * 10
        #df['GARCH_pred'] = df['GARCH_pred'] * 10
        
        ''' original_data 為到時候繪圖用的原始資料RV，既是我們的訓練X也是我們的目標Y'''
        self.original_data = df[['日期', 'daily_RV']]
        
        '''將資料拆成train/test set'''
        split_point = int(len(df) * self.tt_ratio)
        Train_set = df.iloc[0:split_point]
        
        Test_set = df[split_point - self.window_size::] #ex:window_size=7 ，則 train_data 的最後6日 也將會是 test_set 的input
        Test_set.reset_index(drop = True, inplace = True)
        
        self.RV_max = Train_set['daily_RV'].max()
        self.RV_min = Train_set['daily_RV'].min()
        self.IV_max = Train_set['IV加權'].max()
        self.IV_min = Train_set['IV加權'].min()
        
        ''' 依據 有無使用IV & train或test 來調整資料的輸入輸出 '''
        if self.add_IV_input == True: #若為True，表示同時輸入RV跟IV做輸入
            if self.mode == 'test' :
                RV_Test_X, RV_Test_Y =  create_sequences(Test_set['daily_RV'], self.window_size)
                IV_Test_X, IV_Test_Y =  create_sequences(Test_set['IV加權'], self.window_size)
                '''INPUT SCALE'''
                RV_Test_X = (RV_Test_X - self.RV_min)/(self.RV_max - self.RV_min)
                IV_Test_X = (IV_Test_X - self.IV_min)/(self.IV_max - self.IV_min)
                
                combine_array_X = [np.concatenate((RV, IV), axis=1) for RV, IV in zip(RV_Test_X, IV_Test_X)]
                combine_array_X = np.array(combine_array_X)
                self.test_X = torch.FloatTensor(combine_array_X)

            elif self.mode == 'train':

                # RV_train 依據seq_length而有所改變
                RV_train_X, RV_train_Y = create_sequences(Train_set['daily_RV'], self.window_size)
                IV_train_X, IV_train_Y = create_sequences(Train_set['IV加權'], self.window_size)
                '''INPUT SCALE'''
                RV_train_X = (RV_train_X - self.RV_min)/(self.RV_max - self.RV_min)
                IV_train_X = (IV_train_X - self.IV_min)/(self.IV_max - self.IV_min)
                          
                combine_array_X = [np.concatenate((RV, IV), axis=1) for RV, IV in zip(RV_train_X, IV_train_X)]
                combine_array_X = np.array(combine_array_X)
                self.train_X = torch.FloatTensor(combine_array_X)
                self.train_Y = torch.FloatTensor(RV_train_Y)
           
        elif self.add_IV_input == False:
            if self.mode == 'test':
                RV_Test_X, RV_Test_Y =  create_sequences(Test_set['daily_RV'], self.window_size)
                '''INPUT SCALE'''
                RV_Test_X = (RV_Test_X - self.RV_min)/(self.RV_max - self.RV_min)
                self.test_X = torch.FloatTensor(RV_Test_X)

            elif self.mode == 'train':
                 # RV_train 依據seq_length而有所改變
                RV_train_X,RV_train_Y = create_sequences(Train_set['daily_RV'], self.window_size)
                '''INPUT SCALE'''
                RV_train_X = (RV_train_X - self.RV_min)/(self.RV_max - self.RV_min)
                
                self.train_X = torch.FloatTensor(RV_train_X)   
                self.train_Y = torch.FloatTensor(RV_train_Y)
    
    # 加入GARCH 參數
    # 無論有無加上IV 最後都是 self.train_X, self.test_X
        if self.mode == 'test':
            aplha_X, aplha_Y =  create_sequences(Test_set['alpha'], self.window_size)
            beta_X, beta_Y =  create_sequences(Test_set['beta'], self.window_size)
            #GARCH_pred_X, GARCH_pred_Y =  create_sequences(Test_set['GARCH_pred'], self.window_size)
            
            GARCH_X = [np.concatenate((aplha, beta), axis=1) for aplha, beta in zip(aplha_X, beta_X )]
            GARCH_X = np.array(GARCH_X)
            GARCH_X = torch.FloatTensor(GARCH_X)
            
            self.test_X = torch.cat((self.test_X, GARCH_X), 2) # cat == concat 將tensor做合併
        
        elif self.mode == 'train':     
            aplha_X, aplha_Y =  create_sequences(Train_set['alpha'], self.window_size)
            beta_X, beta_Y =  create_sequences(Train_set['beta'], self.window_size)
            #GARCH_pred_X, GARCH_pred_Y = create_sequences(Train_set['GARCH_pred'], self.window_size)
                     
            GARCH_X = [np.concatenate((aplha, beta), axis=1) for aplha, beta in zip(aplha_X, beta_X)]
            GARCH_X = np.array(GARCH_X)
            GARCH_X = torch.FloatTensor(GARCH_X)
            
            self.train_X = torch.cat((self.train_X, GARCH_X), 2)
    
    def __getitem__(self, index):
        
        if self.mode == 'train':
            return self.train_X[index], self.train_Y[index]
        
        elif self.mode == 'test':
            return self.test_X[index]
    
    def __len__(self):
        if self.mode == 'train':
            return len(self.train_X)
        
        elif self.mode == 'test':
            return len(self.test_X)

def prep_dataloader(mode, batch_size, train_test_split_ratio, window_size,add_IV_input, n_jobs=0):
    ''' Generates a dataset, then is put into a dataloader. '''
    dataset = RV_dataset(mode=mode, train_test_split_ratio=train_test_split_ratio, window_size=window_size, add_IV_input= add_IV_input)  # Construct dataset
    dataloader = DataLoader(
        dataset, batch_size,
        shuffle=False, drop_last=True, pin_memory=True)                            # Construct dataloader
    return dataloader

# 建立LSTM model

class LSTM(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers):
        super(LSTM, self).__init__()
        
        self.input_size = input_size #input size
        self.hidden_size = hidden_size #hidden state     
        self.num_layers = num_layers
        
        self.lstm = nn.LSTM(input_size=self.input_size, hidden_size=self.hidden_size, num_layers = self.num_layers,dropout = 0.2, batch_first=True) #lstm
        self.net = nn.Sequential(
            nn.ReLU(),
            nn.Linear(self.hidden_size * self.num_layers, 64),
            nn.ReLU(),
            nn.Linear(64, 32),
            nn.ReLU(),
            nn.Linear(32, 1),
        )
        
    
    def forward(self,x):
        """
        assumes x.shape represents (batch_size, sequence_size, input_size)
        """
        batch_size, sq_len, _ = x.size()
        
        h_0 = torch.zeros(self.num_layers, batch_size, self.hidden_size) #hidden state
        c_0 = torch.zeros(self.num_layers, batch_size, self.hidden_size) #internal state
        nn.init.xavier_normal_(h_0)
        nn.init.xavier_normal_(c_0)
        # Propagate input through LSTM
        output, (hn, cn) = self.lstm(x, (h_0, c_0)) #lstm with input, hidden, and internal state
        hn = hn.view(-1, self.hidden_size * self.num_layers ) #reshaping the data for Dense layer next
        #out = self.relu(hn)
        out = self.net(hn)
        return out
